  InsertionSort(){
    for(i = 1; i < arraySize; i++)
    {
      j = i-1;
      x = A[i];
      while(j > -1 && A[j] > x)
      {
        A[j+1] = A[j];
        j--;
      }
      A[j+1] = x;
    }
  }


  SelectionSort(){
    for(i = 0; i < arraySize-1; i++)
    {
      for(j = k = i; j < arraySize; j++)
      {
        if(A[j] < A[k])
        k = j;
      }
      swap(A[i], A[k]);
    }
  }


  Node* insert(Node* node, int key)
  {
    if (node == NULL) return newNode(key);
    if (key < node->key)
      node->left  = insert(node->left, key);
     else
      node->right = insert(node->right, key);
    return node;
  }

  void lnr(Node *root, int arr[], int &i)
  {
    if (root != NULL)
    {
      lnr(root->left, arr, i);
      arr[i++] = root->key;
      lnr(root->right, arr, i);
    }
  }

  Shellsort(){
    for (int gap = n / 2; gap > 0; gap /= 2)
    {
      for (int i = gap; i < n; ++i)
      {
        for (int j = i; j >= gap; j -= gap)
          if (arr[j - gap] > arr[j]) 
            swap(arr[j - gap], arr[j]);
          else 
            break;;
      }
    }
